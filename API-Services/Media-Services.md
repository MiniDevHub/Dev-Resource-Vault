```javascript
const cloudinary = require("cloudinary").v2;

// Configure

  cloud_name: process.env.CLOUDINARY_CLOUD_NAME,
  api_key: process.env.CLOUDINARY_API_KEY,
  api_secret: process.env.CLOUDINARY_API_SECRET,

// Upload image
async function uploadImage(filePath) {
  try {
    const result = await cloudinary.uploader.upload(filePath, {
      folder: "my-app",
      resource_type: "auto",
      transformation: [
        { width: 1000, crop: "limit" }, // Max width
        { quality: "auto" }, // Automatic quality
        { fetch_format: "auto" }, // Auto format (WebP if supported)
      ],
    });

    return {
      url: result.secure_url,
      publicId: result.public_id,
      format: result.format,
      width: result.width,
      height: result.height,
    };
  } catch (error) {
    console.error("Upload error:", error);
    throw error;
  }
}

// Upload from URL
async function uploadFromURL(imageUrl) {
  const result = await cloudinary.uploader.upload(imageUrl, {
    folder: "my-app",
  });
  return result.secure_url;
}

// Upload with Express multer
const multer = require("multer");
const { CloudinaryStorage } = require("multer-storage-cloudinary");

  cloudinary: cloudinary,
  params: {
    folder: "uploads",
    allowed_formats: ["jpg", "jpeg", "png", "gif", "webp"],
  },
});

const upload = multer({ storage: storage });

app.post("/upload", upload.single("image"), (req, res) => {
  res.json({
    url: req.file.path,
    publicId: req.file.filename,
  });
});

  return cloudinary.url(publicId, transformations);
}
```

```javascript
const sharp = require("sharp");

    { width: 1280, suffix: "large" },
  ];

// Complete image processing pipeline
async function processUploadedImage(file) {
  const filename = `${Date.now()}-${file.originalname}`;
  const inputPath = file.path;

  // Create multiple versions
  const versions = {
    original: `uploads/original/${filename}`,
    thumbnail: `uploads/thumb/${filename}`,
    medium: `uploads/medium/${filename}`,
    webp: `uploads/webp/${filename.replace(/\.[^.]+$/, ".webp")}`,
  };

  // Original (optimized)
  await sharp(inputPath)
    .jpeg({ quality: 85, progressive: true })
    .toFile(versions.original);

  // Thumbnail
  await sharp(inputPath)
    .resize(150, 150, { fit: "cover" })
    .jpeg({ quality: 80 })
    .toFile(versions.thumbnail);

  // Medium
  await sharp(inputPath)
    .resize(800)
    .jpeg({ quality: 80 })
    .toFile(versions.medium);

  // WebP version
  await sharp(inputPath).webp({ quality: 80 }).toFile(versions.webp);

  // Clean up temp file
  fs.unlinkSync(inputPath);

  return versions;
}
```

```javascript
const imageObserver = new IntersectionObserver((entries, observer) => {
  entries.forEach((entry) => {
    if (entry.isIntersecting) {
      const img = entry.target;
      img.src = img.dataset.src;
```

Complete image upload example:

```javascript
// Complete image upload with optimization
const express = require("express");
const multer = require("multer");
const sharp = require("sharp");
const cloudinary = require("cloudinary").v2;
const app = express();

// Configure Cloudinary
cloudinary.config({
  cloud_name: process.env.CLOUDINARY_CLOUD_NAME,
  api_key: process.env.CLOUDINARY_API_KEY,
  api_secret: process.env.CLOUDINARY_API_SECRET,
});

// Multer memory storage
const upload = multer({
  storage: multer.memoryStorage(),
  limits: {
    fileSize: 5 * 1024 * 1024, // 5MB limit
  },
  fileFilter: (req, file, cb) => {
    // Accept images only
    if (!file.mimetype.startsWith("image/")) {
      return cb(new Error("Only images allowed"), false);
    }
    cb(null, true);
  },
});

// Upload endpoint
app.post("/upload", upload.single("image"), async (req, res) => {
  try {
    if (!req.file) {
      return res.status(400).json({ error: "No file uploaded" });
    }

    // Process image with Sharp
    const processedImage = await sharp(req.file.buffer)
      .resize(2000, 2000, {
        fit: "inside",
        withoutEnlargement: true,
      })
      .jpeg({ quality: 85, progressive: true })
      .toBuffer();

    // Upload to Cloudinary
    const uploadStream = cloudinary.uploader.upload_stream(
      {
        folder: "uploads",
        resource_type: "auto",
        transformation: [{ quality: "auto" }, { fetch_format: "auto" }],
      },
      (error, result) => {
        if (error) {
          return res.status(500).json({ error: error.message });
        }

        // Generate various sizes
        const urls = {
          original: result.secure_url,
          thumbnail: cloudinary.url(result.public_id, {
            width: 150,
            height: 150,
            crop: "fill",
          }),
          medium: cloudinary.url(result.public_id, {
            width: 800,
            height: 600,
            crop: "limit",
          }),
        };

        res.json({
          success: true,
          urls: urls,
          publicId: result.public_id,
          format: result.format,
          size: result.bytes,
        });
      }
    );

    // Pipe processed image to Cloudinary
    require("stream").Readable.from(processedImage).pipe(uploadStream);
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

// Delete image
app.delete("/image/:publicId", async (req, res) => {
  try {
    await cloudinary.uploader.destroy(req.params.publicId);
    res.json({ success: true });
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

app.listen(3000, () => console.log("Server running on port 3000"));
```
