```javascript
  to: "recipient@example.com",
```

```python

import os
from sendgrid import SendGridAPIClient
from sendgrid.helpers.mail import Mail

message = Mail(
    from_email='sender@yourdomain.com',
    to_emails='recipient@example.com',
    subject='Hello from SendGrid!',
    html_content='<strong>This is a test email</strong>'
)

try:
    sg = SendGridAPIClient(os.environ.get('SENDGRID_API_KEY'))
    response = sg.send(message)
    print(f"Status Code: {response.status_code}")
except Exception as e:
    print(e)
```

```javascript
function isValidEmail(email) {
  const re = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  return re.test(email);
}

// Check disposable domains
const disposableDomains = [
  "tempmail.com",
  "10minutemail.com",
  "guerrillamail.com",
  // ... more
];

function isDisposable(email) {
  const domain = email.split("@")[1];
  return disposableDomains.includes(domain.toLowerCase());
}

// Full validation
function validateEmail(email) {
  if (!isValidEmail(email)) {
    return { valid: false, reason: "Invalid format" };
  }

  if (isDisposable(email)) {
    return { valid: false, reason: "Disposable email" };
  }

  return { valid: true };
}

// Usage
const result = validateEmail("user@example.com");
if (!result.valid) {
  console.error(result.reason);
}
```

```javascript
class RateLimiter {
  constructor(maxPerMinute) {
    this.maxPerMinute = maxPerMinute;
    this.queue = [];
  }

  async add(fn) {
    // Remove old entries
    const now = Date.now();
    this.queue = this.queue.filter((time) => now - time < 60000);

    // Check limit
    if (this.queue.length >= this.maxPerMinute) {
      const waitTime = 60000 - (now - this.queue[0]);
      await new Promise((resolve) => setTimeout(resolve, waitTime));
      return this.add(fn);
    }

    this.queue.push(now);
    return fn();
  }
}
```

```javascript
async function sendEmailWithRetry(emailData, maxRetries = 3) {
  for (let attempt = 1; attempt <= maxRetries; attempt++) {
    try {
      const result = await sendEmail(emailData);
      console.log(`Email sent successfully on attempt ${attempt}`);
      return result;
    } catch (error) {
      console.error(`Attempt ${attempt} failed:`, error.message);

      // Don't retry on these errors
      if (error.code === 400 || error.code === 401) {
        throw new Error(`Permanent error: ${error.message}`);
      }

      // Wait before retry (exponential backoff)
      if (attempt < maxRetries) {
        const waitTime = Math.pow(2, attempt) * 1000;
        console.log(`Waiting ${waitTime}ms before retry...`);
        await new Promise((resolve) => setTimeout(resolve, waitTime));
      }
    }
  }

  throw new Error(`Failed to send email after ${maxRetries} attempts`);
}

// Usage

  await sendEmailWithRetry(emailData);
} catch (error) {
  // Log to error tracking service
  console.error("Failed to send email:", error);
  // Maybe queue for later retry
```
